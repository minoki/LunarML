Compile and run the examples:

```console
$ lunarml compile commandline.sml
$ lua commandline.lua foo bar

$ lunarml compile date.sml
$ lua date.lua

$ lunarml compile --nodejs-cps delim-cont.mlb
$ node delim-cont.mjs

$ lunarml compile fib.sml
$ lua fib.lua

$ lunarml compile fizzbuzz.sml
$ lua fizzbuzz.lua

$ lunarml compile --nodejs hello-js.mlb
$ node hello-js.mjs

$ lunarml compile hello.sml
$ lua hello.lua

$ lunarml compile listrev.sml
$ lua listrev.lua

$ lunarml compile --nodejs-cps monad.mlb
$ node monad.mjs

$ lunarml compile pipe.mlb
$ lua pipe.lua
```
